public class Program
{
    static void Main()
    {
        {
            Console.InputEncoding = Encoding.Unicode;
            Console.OutputEncoding = Encoding.Unicode;

            Console.Write("Введіть кількість елементів масиву n: ");
            int n = int.Parse(Console.ReadLine());

            double[] arr = new double[n];

           
            Console.WriteLine("Введіть елементи масиву:");
            for (int i = 0; i < n; i++)
            {
                Console.Write($"arr[{i}] = ");
                arr[i] = double.Parse(Console.ReadLine());
            }

           
            double sumNeg = 0;
            foreach (double x in arr)
            {
                if (x < 0)
                    sumNeg += x;
            }

            int minIndex = 0, maxIndex = 0;
            for (int i = 1; i < n; i++)
            {
                if (arr[i] < arr[minIndex]) minIndex = i;
                if (arr[i] > arr[maxIndex]) maxIndex = i;
            }

            int start = Math.Min(minIndex, maxIndex) + 1;
            int end = Math.Max(minIndex, maxIndex) - 1;

            double product = 1;
            if (start <= end)
            {
                for (int i = start; i <= end; i++)
                    product *= arr[i];
            }
            else
            {
                product = 0; 
            }

            Array.Sort(arr);

            Console.WriteLine($"\nСума негативних елементів: {sumNeg}");
            Console.WriteLine($"Добуток елементів між мінімальним і максимальним: {product}");
            Console.WriteLine("Масив після сортування за зростанням:");
            foreach (double x in arr)
                Console.Write(x + " ");
            Console.WriteLine();
        }
    }



}
